\section{Trigonometry}

\subsection{Period}
$\sin a = \sin (a + 2 k \pi)$\\
$\cos a = \cos (a + 2 k \pi)$\\
$\tan a = \tan (a + k \pi)$\\
$\cot a = \cot (a + k \pi)$\\

\subsection{$a \pm b$}
$\sin (a \pm b) = \sin a \cos b \pm\cos a  \sin a$\\
$\cos(a \pm b) = \cos a \cos b \mp \sin a \sin b$\\
$\sin 2 a = 2 \sin a \cos a$\\
$\cos 2 a 
= \cos^{2} a - \sin^{2} a 
= 1 - 2 \sin^{2} a
= 2 \cos^{2} a - 1$\\
$\tan (a \pm b) = \frac{\tan a + \tan b}{1 \mp \tan a \tan b}$\\



\subsection{odd, even}
$\sin (-a) = - \sin a$\\
$\cos (-a) = \cos a$\\
$\tan (-a) = - \tan a$\\

\subsection{genel}
$\sin^{2} a + \cos^{2} a = 1$\\
$\tan a \cot a = 1$\\
